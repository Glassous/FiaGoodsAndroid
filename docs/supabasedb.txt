环境变量：
SUPABASE_URL=https://<your-project-id>.supabase.co
SUPABASE_ANON_KEY=<your-anon-or-publishable-key>

表结构：
create table if not exists public.cargo_items (
  id uuid primary key,
  name text not null,
  category text not null,
  stock integer,
  price numeric(12,2) not null default 0,
  sold integer not null default 0,
  brief text not null default '',
  description text not null default '',
  specs text not null default '',
  image_urls jsonb not null default '[]'::jsonb
);

alter table public.cargo_items enable row level security;

create policy "anon_select" on public.cargo_items
  for select using (true);

create policy "anon_insert" on public.cargo_items
  for insert with check (true);

create policy "anon_update" on public.cargo_items
  for update using (true) with check (true);

create policy "anon_delete" on public.cargo_items
  for delete using (true);
